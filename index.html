★ Foundation
Arrays
Contiguous memory blocks storing same-type elements. The backbone of nearly every algorithm — master this and everything else follows. Enables O(1) index access and cache-friendly iteration.
Concept
Implementation
Use Cases

Memory Layout

Elements at contiguous addresses. arr[i] = base_addr + i × element_size. Direct index access = O(1) regardless of array size.

Static vs Dynamic

Static: fixed size at compile time. Dynamic (Python list): doubles capacity when full — amortized O(1) append via doubling strategy.

Two Pointer Technique

Left + right pointers moving toward each other. Solves sorted array problems in O(n) with O(1) space. Essential pattern for interviews.

Sliding Window

Expand right, shrink left to maintain a valid window. Solves subarray/substring problems in O(n). Avoids O(n²) nested loops.

# ── Two Sum (sorted array) — O(n) time, O(1) space ──────────────────
def two_sum_sorted(arr, target):
    left, right = 0, len(arr) - 1
    while left < right:
        s = arr[left] + arr[right]
        if s == target:   return [left, right]
        elif s < target:  left += 1
        else:             right -= 1
    return []

# ── Sliding Window Maximum — O(n) using deque ──────────────────────
from collections import deque
def sliding_window_max(nums, k):
    dq, res = deque(), []
    for i, n in enumerate(nums):
        while dq and nums[dq[-1]] < n: dq.pop()
        dq.append(i)
        if dq[0] == i - k:          dq.popleft()
        if i >= k - 1:              res.append(nums[dq[0]])
    return res

# ── Kadane's Algorithm (Max Subarray) — O(n) ───────────────────────
def max_subarray(nums):
    max_sum = cur = nums[0]
    for n in nums[1:]:
        cur = max(n, cur + n)
        max_sum = max(max_sum, cur)
    return max_sum

# ── Prefix Sum — range query O(1) after O(n) build ─────────────────
def build_prefix(arr):
    prefix = [0] * (len(arr) + 1)
    for i, v in enumerate(arr):
        prefix[i+1] = prefix[i] + v
    return prefix

def range_sum(prefix, l, r):     # inclusive [l, r]
    return prefix[r+1] - prefix[l]

# ── Binary Search — all variants ───────────────────────────────────
def binary_search(arr, target):
    lo, hi = 0, len(arr) - 1
    while lo <= hi:
        mid = lo + (hi - lo) // 2   # avoids overflow
        if   arr[mid] == target: return mid
        elif arr[mid] < target:  lo = mid + 1
        else:                    hi = mid - 1
    return -1

def lower_bound(arr, target):     # leftmost occurrence
    lo, hi, ans = 0, len(arr) - 1, -1
    while lo <= hi:
        mid = lo + (hi - lo) // 2
        if arr[mid] >= target: ans = mid; hi = mid - 1
        else: lo = mid + 1
    return ans

# ── Dutch National Flag (3-way partition) ──────────────────────────
def sort_012(arr):
    lo = mid = 0; hi = len(arr) - 1
    while mid <= hi:
        if   arr[mid] == 0: arr[lo], arr[mid] = arr[mid], arr[lo]; lo += 1; mid += 1
        elif arr[mid] == 1: mid += 1
        else:                arr[mid], arr[hi] = arr[hi], arr[mid]; hi -= 1
    return arr

Pattern Matching
Strings
Immutable sequences of characters with rich algorithm space: sliding windows, KMP for linear pattern matching, Rabin-Karp rolling hashes, tries, and Manacher's O(n) palindrome detection.
Concept
Implementation
Use Cases
⚡ Strings in most languages are immutable. String concatenation in a loop is O(n²) — always use a list and ''.join() or a StringBuilder for building strings iteratively.

Sliding Window

Maintain a window of characters, expand right and shrink left. Solves longest substring without repeat, min window substring in O(n).

KMP Algorithm

Failure function (LPS array) avoids re-scanning. Pattern matching in O(n+m) vs O(n×m) naive. Used in grep, text editors, DNA sequencing.

Rabin-Karp

Rolling hash computes substring hash in O(1). Excellent for multiple pattern search. Average O(n+m). Used in plagiarism detection and duplicate detection.

Manacher's Algorithm

Finds all palindromic substrings in O(n) by reusing previously computed palindrome centers. Transforms odd/even lengths into a unified problem.

# ── Longest Substring Without Repeating — O(n) ─────────────────────
def length_of_longest(s):
    seen, start, ans = {}, 0, 0
    for i, c in enumerate(s):
        if c in seen and seen[c] >= start:
            start = seen[c] + 1
        seen[c] = i
        ans = max(ans, i - start + 1)
    return ans

# ── KMP Pattern Search — O(n+m) ────────────────────────────────────
def kmp_search(text, pattern):
    def build_lps(p):
        lps = [0] * len(p); length = 0; i = 1
        while i < len(p):
            if p[i] == p[length]:
                length += 1; lps[i] = length; i += 1
            elif length: length = lps[length-1]
            else: lps[i] = 0; i += 1
        return lps
    lps = build_lps(pattern); i = j = 0; matches = []
    while i < len(text):
        if text[i] == pattern[j]: i += 1; j += 1
        if j == len(pattern): matches.append(i - j); j = lps[j-1]
        elif i < len(text) and text[i] != pattern[j]:
            if j: j = lps[j-1]
            else: i += 1
    return matches

# ── Minimum Window Substring — O(n) ────────────────────────────────
from collections import Counter
def min_window(s, t):
    need = Counter(t); have = {}
    formed = required = len(need)
    l, ans = 0, (float('inf'), 0, 0)
    for r, c in enumerate(s):
        have[c] = have.get(c, 0) + 1
        if c in need and have[c] == need[c]: formed -= 1
        while formed == 0:
            if r - l + 1 < ans[0]: ans = (r - l + 1, l, r)
            lc = s[l]; have[lc] -= 1; l += 1
            if lc in need and have[lc] < need[lc]: formed += 1
    return s[ans[1]:ans[2]+1] if ans[0] != float('inf') else ""

# ── Longest Palindromic Substring — O(n) Manacher's ────────────────
def longest_palindrome(s):
    t = '#' + '#'.join(s) + '#'    # unify odd/even
    n = len(t); p = [0] * n; c = r = 0
    for i in range(n):
        m = 2*c - i
        if i < r: p[i] = min(r - i, p[m])
        while i+p[i]+1 < n and i-p[i]-1 >= 0 and t[i+p[i]+1] == t[i-p[i]-1]: p[i] += 1
        if i + p[i] > r: c, r = i, i + p[i]
    best = max(range(n), key=lambda x: p[x])
    start = (best - p[best]) // 2
    return s[start:start + p[best]]

Mind Bender
Recursion
A function that calls itself with a smaller input. Every recursive problem needs a base case (termination) and a recursive case that converges toward it. Powers divide-and-conquer, backtracking, and all tree/graph traversal.
Concept
Implementation
Use Cases
⚠️ Every recursive call uses stack space. Python's default limit is 1000 frames. For deep recursion, use iterative + explicit stack or sys.setrecursionlimit() carefully. Tail call optimization is NOT supported in Python.

Call Stack Model

Each call pushes a frame onto the stack. Returns pop the frame. For fib(5): 15 calls, max depth 5. Stack overflow = recursion too deep.

Memoization (Top-Down DP)

Cache results of subproblems with @lru_cache. Converts exponential O(2ⁿ) recursion to O(n). Each unique subproblem computed exactly once.

Backtracking Pattern

Choose → Recurse → Undo. Try a choice, go deeper, undo if it fails. Explores all valid paths. Prune invalid branches early for performance.

Divide & Conquer

Split into independent subproblems, solve recursively, combine results. Merge Sort, Quick Sort follow T(n) = 2T(n/2) + O(n) → O(n log n).

from functools import lru_cache

# ── Fibonacci: naive vs memoized ────────────────────────────────────
def fib_naive(n):                    # O(2^n) — DO NOT USE for large n
    if n <= 1: return n
    return fib_naive(n-1) + fib_naive(n-2)

@lru_cache(maxsize=None)
def fib_memo(n):                     # O(n) — each n computed exactly once
    if n <= 1: return n
    return fib_memo(n-1) + fib_memo(n-2)

# ── Backtracking Template — All Permutations ────────────────────────
def permutations(nums):
    res = []
    def backtrack(path, used):
        if len(path) == len(nums):
            res.append(path[:]); return
        for i, n in enumerate(nums):
            if used[i]: continue
            used[i] = True; path.append(n)
            backtrack(path, used)
            path.pop(); used[i] = False  # UNDO — key step
    backtrack([], [False] * len(nums))
    return res

# ── N-Queens — classic backtracking ────────────────────────────────
def n_queens(n):
    res, cols = [], set()
    diag1, diag2 = set(), set()
    def bt(row, board):
        if row == n: res.append(board[:]); return
        for col in range(n):
            if col in cols or row-col in diag1 or row+col in diag2: continue
            cols.add(col); diag1.add(row-col); diag2.add(row+col)
            board.append('.'*col + 'Q' + '.'*(n-col-1))
            bt(row+1, board)
            board.pop(); cols.discard(col); diag1.discard(row-col); diag2.discard(row+col)
    bt(0, [])
    return res

# ── Merge Sort — Divide & Conquer O(n log n) ────────────────────────
def merge_sort(arr):
    if len(arr) <= 1: return arr
    mid = len(arr) // 2
    L = merge_sort(arr[:mid]); R = merge_sort(arr[mid:])
    res = []; i = j = 0
    while i < len(L) and j < len(R):
        if L[i] <= R[j]: res.append(L[i]); i += 1
        else:            res.append(R[j]); j += 1
    return res + L[i:] + R[j:]

# ── Quick Sort — O(n log n) average, O(n²) worst ───────────────────
def quick_sort(arr, lo=0, hi=None):
    if hi is None: hi = len(arr) - 1
    if lo < hi:
        p = _partition(arr, lo, hi)
        quick_sort(arr, lo, p-1)
        quick_sort(arr, p+1, hi)

def _partition(arr, lo, hi):
    pivot = arr[hi]; i = lo - 1
    for j in range(lo, hi):
        if arr[j] <= pivot:
            i += 1; arr[i], arr[j] = arr[j], arr[i]
    arr[i+1], arr[hi] = arr[hi], arr[i+1]
    return i + 1

# ── Heap Sort — O(n log n) guaranteed, in-place ─────────────────────
def heap_sort(arr):
    n = len(arr)
    for i in range(n//2-1, -1, -1):  # build max-heap
        _heapify(arr, n, i)
    for i in range(n-1, 0, -1):
        arr[0], arr[i] = arr[i], arr[0]   # max to end
        _heapify(arr, i, 0)

def _heapify(arr, n, i):
    largest = i; l, r = 2*i+1, 2*i+2
    if l < n and arr[l] > arr[largest]: largest = l
    if r < n and arr[r] > arr[largest]: largest = r
    if largest != i:
        arr[i], arr[largest] = arr[largest], arr[i]
        _heapify(arr, n, largest)

# ── Counting Sort — O(n+k), integers only ──────────────────────────
def counting_sort(arr):
    k = max(arr); count = [0] * (k + 1)
    for x in arr: count[x] += 1
    for i in range(1, k+1): count[i] += count[i-1]  # prefix sum
    out = [0] * len(arr)
    for x in reversed(arr):
        count[x] -= 1; out[count[x]] = x
    return out

# ── Binary Search on Answer — O(n log(max-min)) ────────────────────
# Problem: split n books into m students, minimize max pages per student
def book_allocation(books, m):
    def feasible(max_pages):
        students = pages = 0
        for b in books:
            if pages + b > max_pages: students += 1; pages = 0
            pages += b
        return students + 1 <= m
    lo, hi = max(books), sum(books); ans = hi
    while lo <= hi:
        mid = (lo + hi) // 2
        if feasible(mid): ans = mid; hi = mid - 1
        else: lo = mid + 1
    return ans

Floyd's Cycle Detection

Slow pointer moves 1 step, fast moves 2. If they meet, cycle exists. O(n) time, O(1) space. Mathematical proof also finds the cycle entry point.

Two Pointer Tricks

Find middle: slow+fast together. kth from end: advance lead k steps first, then move both. Merge sorted lists: compare heads, advance the smaller.

class ListNode:
    def __init__(self, val=0, nxt=None):

# ── Reverse Linked List — O(n) time, O(1) space ─────────────────────
def reverse(head):
    prev = None
    while head:
        nxt = head.next
        head.next = prev
        prev, head = head, nxt
    return prev

# ── Floyd's Cycle Detection ──────────────────────────────────────────
def has_cycle(head):
    slow = fast = head
    while fast and fast.next:
        slow = slow.next; fast = fast.next.next
        if slow is fast: return True
    return False

# ── Find cycle entry point (mathematical proof) ──────────────────────
def cycle_start(head):
    slow = fast = head
    while fast and fast.next:
        slow = slow.next; fast = fast.next.next
        if slow is fast:
            slow2 = head
            while slow is not slow2:
                slow = slow.next; slow2 = slow2.next
            return slow   # entry point
    return None

# ── Merge Two Sorted Lists — O(n+m) ─────────────────────────────────
def merge_sorted(l1, l2):
    dummy = cur = ListNode(0)
    while l1 and l2:
        if l1.val <= l2.val: cur.next = l1; l1 = l1.next
        else:               cur.next = l2; l2 = l2.next
        cur = cur.next
    cur.next = l1 or l2
    return dummy.next

# ── LRU Cache — O(1) get/put via doubly-LL + hashmap ────────────────
from collections import OrderedDict
class LRUCache:
    def __init__(self, cap):
        self.cap = cap; self.cache = OrderedDict()
    def get(self, key):
        if key not in self.cache: return -1
        self.cache.move_to_end(key)
        return self.cache[key]
    def put(self, key, val):
        if key in self.cache: self.cache.move_to_end(key)
        self.cache[key] = val
        if len(self.cache) > self.cap:
            self.cache.popitem(False)   # evict LRU (first item)

🔑 Monotonic Stack Pattern: maintains elements in strictly increasing/decreasing order. Solves "next greater element" and "largest rectangle in histogram" in O(n) — problems that look O(n²) at first glance.

Monotonic Stack

Pop elements that violate the monotonic property when inserting. After processing, the stack contains elements in order. Solves NGE, stock span, trapping rain water.

Deque (Double-ended Queue)

O(1) push/pop from both ends. Python's collections.deque. Powers sliding window maximum and BFS with level tracking.

from collections import deque

# ── Valid Parentheses — O(n) ─────────────────────────────────────────
def is_valid(s):
    stack = []; pairs = {')':'(', '}':'{', ']':'['}
    for c in s:
        if c in '([{': stack.append(c)
        elif not stack or stack[-1] != pairs[c]: return False
        else: stack.pop()
    return not stack

# ── Next Greater Element — monotonic stack O(n) ──────────────────────
def next_greater(nums):
    res = [-1] * len(nums); stack = []
    for i, n in enumerate(nums):
        while stack and nums[stack[-1]] < n:
            res[stack.pop()] = n
        stack.append(i)
    return res

# ── Largest Rectangle in Histogram — O(n) ───────────────────────────
def largest_rect(heights):
    stack = []; max_area = 0
    heights = heights + [0]            # sentinel forces stack drain
    for i, h in enumerate(heights):
        start = i
        while stack and stack[-1][1] > h:
            idx, ht = stack.pop()
            max_area = max(max_area, ht * (i - idx))
            start = idx
        stack.append((start, h))

class MinStack:
    def __init__(self): self.stack = []; self.min_stack = []
    def push(self, x):
        self.stack.append(x)
        m = min(x, self.min_stack[-1] if self.min_stack else x)
        self.min_stack.append(m)
    def pop(self): self.stack.pop(); self.min_stack.pop()
    def get_min(self): return self.min_stack[-1]

# ── BFS using Queue ──────────────────────────────────────────────────
def bfs(graph, start):
    visited, queue, order = {start}, deque([start]), []
    while queue:
        node = queue.popleft(); order.append(node)
        for nbr in graph.get(node, []):
            if nbr not in visited:
                visited.add(nbr); queue.append(nbr)
    return order

import hashlib, bisect
class ConsistentHash:
    def __init__(self, replicas=100):
        self.ring = {}; self.replicas = replicas; self.keys = []
    def _h(self, key):
        return int(hashlib.md5(key.encode()).hexdigest(), 16) % (2**32)
    def add_node(self, node):
        for i in range(self.replicas):
            h = self._h(f"{node}:{i}")
            self.ring[h] = node; bisect.insort(self.keys, h)
    def get_node(self, key):
        h = self._h(key)
        idx = bisect.bisect(self.keys, h) % len(self.keys)
        return self.ring[self.keys[idx]]

def lca(root, p, q):

class Trie:
    def __init__(self): self.root = {}
    def insert(self, word):
        node = self.root
        for c in word: node = node.setdefault(c, {})
        node['#'] = True              # end-of-word marker
    def search(self, word):
        node = self.root
        for c in word:
            if c not in node: return False
            node = node[c]
        return '#' in node
    def starts_with(self, prefix):
        node = self.root
        for c in prefix:
            if c not in node: return False
            node = node[c]
        return True

# ── Segment Tree — O(log n) range sum + point update ─────────────────
class SegTree:
    def __init__(self, arr):
        self.n = len(arr); self.tree = [0] * (2*self.n)
        for i, v in enumerate(arr): self.tree[self.n+i] = v
        for i in range(self.n-1, 0, -1):
            self.tree[i] = self.tree[2*i] + self.tree[2*i+1]
    def update(self, i, val):
        i += self.n; self.tree[i] = val
        while i > 1: i //= 2; self.tree[i] = self.tree[2*i]+self.tree[2*i+1]
    def query(self, l, r):              # sum in [l, r)
        res = 0; l += self.n; r += self.n
        while l < r:
            if l&1: res += self.tree[l]; l += 1
            if r&1: r -= 1; res += self.tree[r]
            l >>= 1; r >>= 1
        return res

Representation

from collections import defaultdict, deque
import heapq

# ── Dijkstra's Shortest Path — O((V+E) log V) ────────────────────────
def dijkstra(graph, start, n):
    dist = [float('inf')] * n; dist[start] = 0
    heap = [(0, start)]
    while heap:
        d, u = heapq.heappop(heap)
        if d > dist[u]: continue          # stale entry
        for v, w in graph[u]:
            if dist[u] + w < dist[v]:
                dist[v] = dist[u] + w
                heapq.heappush(heap, (dist[v], v))
    return dist

# ── Topological Sort — Kahn's BFS ────────────────────────────────────
def topo_sort(n, edges):
    graph = defaultdict(list); indegree = [0] * n
    for u, v in edges: graph[u].append(v); indegree[v] += 1
    q = deque(i for i in range(n) if indegree[i] == 0)
    order = []
    while q:
        u = q.popleft(); order.append(u)
        for v in graph[u]:
            indegree[v] -= 1
            if indegree[v] == 0: q.append(v)
    return order if len(order) == n else []    # [] = cycle detected

# ── Union-Find (DSU) with path compression + rank ────────────────────
class DSU:
    def __init__(self, n):
        self.p = list(range(n)); self.rank = [0] * n
    def find(self, x):
        if self.p[x] != x:
            self.p[x] = self.find(self.p[x])    # path compression
        return self.p[x]
    def union(self, x, y):
        rx, ry = self.find(x), self.find(y)
        if rx == ry: return False               # already connected
        if self.rank[rx] < self.rank[ry]: rx, ry = ry, rx
        self.p[ry] = rx
        if self.rank[rx] == self.rank[ry]: self.rank[rx] += 1
        return True

# ── DFS — iterative (avoids Python recursion limit) ──────────────────
def dfs(graph, start):
    visited, stack, order = set(), [start], []
    while stack:
        node = stack.pop()
        if node in visited: continue
        visited.add(node); order.append(node)
        for nbr, _ in graph.get(node, []): stack.append(nbr)
    return order

# ── Number of Islands — DFS on grid ──────────────────────────────────
def num_islands(grid):
    def dfs(r, c):
        if r < 0 or r >= len(grid) or c < 0 or c >= len(grid[0]): return
        if grid[r][c] != '1': return
        grid[r][c] = '0'     # mark visited
        dfs(r+1,c); dfs(r-1,c); dfs(r,c+1); dfs(r,c-1)
    count = 0
    for r in range(len(grid)):
        for c in range(len(grid[0])):
            if grid[r][c] == '1': dfs(r, c); count += 1
    return count

Dynamic Programming
Break problems into overlapping subproblems. Store results to avoid recomputation. DP = optimal substructure + overlapping subproblems. Top-down (memoization) or bottom-up (tabulation). Converts exponential to polynomial.
Concept
Implementation
Use Cases
🧠 The DP mindset: "Can I express the answer to a larger version using answers to smaller versions of the same problem?" If yes, you have optimal substructure → DP applies.

1D DP Patterns

dp[i] = best answer considering first i elements. Fibonacci, climbing stairs, house robber, jump game. Transition uses dp[i-1], dp[i-2], or previous elements.

2D DP Patterns

dp[i][j] = answer for subproblem (i, j). LCS, edit distance, 0/1 knapsack, coin change, unique paths. Build from (0,0) to (n,m).

Top-Down vs Bottom-Up

Top-down (memoized recursion): natural to write, only computes needed states. Bottom-up (tabulation): no stack overhead, cache-friendly, easier space optimization.

Space Optimization

Many 2D DP problems need only previous row. Reduce O(nm) → O(n) space. 0/1 Knapsack: iterate right-to-left over capacity to avoid counting same item twice.

# ── 0/1 Knapsack — O(nW) ────────────────────────────────────────────
def knapsack(weights, values, W):
    n = len(weights)
    dp = [[0]*(W+1) for _ in range(n+1)]
    for i in range(1, n+1):
        for w in range(W+1):
            dp[i][w] = dp[i-1][w]             # skip item i
            if weights[i-1] <= w:
                dp[i][w] = max(dp[i][w],
                    dp[i-1][w - weights[i-1]] + values[i-1])  # take it
    return dp[n][W]

# ── Longest Common Subsequence — O(nm) ──────────────────────────────
def lcs(s, t):
    m, n = len(s), len(t)
    dp = [[0]*(n+1) for _ in range(m+1)]
    for i in range(1, m+1):
        for j in range(1, n+1):
            if s[i-1] == t[j-1]: dp[i][j] = dp[i-1][j-1] + 1
            else: dp[i][j] = max(dp[i-1][j], dp[i][j-1])
    return dp[m][n]

# ── Edit Distance (Levenshtein) — O(nm) ────────────────────────────
def edit_distance(s, t):
    m, n = len(s), len(t)
    dp = [[0]*(n+1) for _ in range(m+1)]
    for i in range(m+1): dp[i][0] = i
    for j in range(n+1): dp[0][j] = j
    for i in range(1, m+1):
        for j in range(1, n+1):
            if s[i-1] == t[j-1]: dp[i][j] = dp[i-1][j-1]
            else: dp[i][j] = 1 + min(dp[i-1][j], dp[i][j-1], dp[i-1][j-1])
    return dp[m][n]

# ── Coin Change — O(n·amount) ────────────────────────────────────────
def coin_change(coins, amount):
    dp = [float('inf')] * (amount+1); dp[0] = 0
    for a in range(1, amount+1):
        for c in coins:
            if c <= a: dp[a] = min(dp[a], dp[a-c] + 1)
    return dp[amount] if dp[amount] != float('inf') else -1

# ── Longest Increasing Subsequence — O(n log n) ──────────────────────
import bisect
def lis(nums):
    tails = []
    for n in nums:
        pos = bisect.bisect_left(tails, n)
        if pos == len(tails): tails.append(n)
        else: tails[pos] = n
    return len(tails)

# ── Unique Paths (grid DP) — O(mn) ───────────────────────────────────
def unique_paths(m, n):
    dp = [[1]*n for _ in range(m)]
    for i in range(1, m):
        for j in range(1, n):
            dp[i][j] = dp[i-1][j] + dp[i][j-1]
    return dp[m-1][n-1]
